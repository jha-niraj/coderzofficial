// AI/ML Models
model PredictionModel {
  id          String   @id @default(cuid())
  type        PredictionType
  version     String
  accuracy    Float
  trainedAt   DateTime
  parameters  Json
  isActive    Boolean  @default(false)
  
  predictions StudentPrediction[]
  
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

enum PredictionType {
  AT_RISK
  PLACEMENT_SUCCESS
  SKILL_GAP
  ENGAGEMENT
}

model StudentPrediction {
  id              String   @id @default(cuid())
  studentId       String
  modelId         String
  
  score           Float
  confidence      Float
  factors         Json
  recommendations Json
  
  student         User     @relation(fields: [studentId], references: [id])
  model           PredictionModel @relation(fields: [modelId], references: [id])
  
  createdAt       DateTime @default(now())
  
  @@index([studentId, createdAt])
}

// Campus Drives
model CampusDrive {
  id              String   @id @default(cuid())
  companyLinkId   String
  
  title           String
  startDate       DateTime
  endDate         DateTime
  location        String?
  mode            DriveMode
  
  eligibility     Json     // {year: [], department: [], minCGPA: 7.0}
  roles           Json     // [{title, package, count}]
  
  status          DriveStatus @default(PLANNING)
  
  companyLink     CompanyUniversityLink @relation(fields: [companyLinkId], references: [id])
  
  rounds          DriveRound[]
  applications    DriveApplication[]
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}

enum DriveMode {
  ON_CAMPUS
  VIRTUAL
  HYBRID
}

enum DriveStatus {
  PLANNING
  APPLICATIONS_OPEN
  SHORTLISTING
  IN_PROGRESS
  OFFERS_RELEASED
  COMPLETED
  CANCELLED
}

model DriveRound {
  id              String   @id @default(cuid())
  driveId         String
  
  roundNumber     Int
  type            RoundType
  title           String
  scheduledDate   DateTime
  duration        Int      // minutes
  
  status          RoundStatus @default(PENDING)
  
  drive           CampusDrive @relation(fields: [driveId], references: [id])
  
  schedules       InterviewSchedule[]
  
  @@unique([driveId, roundNumber])
}

enum RoundType {
  ONLINE_ASSESSMENT
  TECHNICAL_INTERVIEW
  HR_INTERVIEW
  GROUP_DISCUSSION
  CASE_STUDY
}

enum RoundStatus {
  PENDING
  IN_PROGRESS
  COMPLETED
}

model InterviewSchedule {
  id              String   @id @default(cuid())
  roundId         String
  applicationId   String
  
  scheduledAt     DateTime
  meetingLink     String?
  room            String?
  panel           String?
  
  status          InterviewStatus @default(SCHEDULED)
  feedback        Json?
  rating          Int?
  
  round           DriveRound @relation(fields: [roundId], references: [id])
  application     DriveApplication @relation(fields: [applicationId], references: [id])
  
  @@index([roundId, scheduledAt])
}

enum InterviewStatus {
  SCHEDULED
  IN_PROGRESS
  COMPLETED
  NO_SHOW
  RESCHEDULED
}

model DriveApplication {
  id              String   @id @default(cuid())
  driveId         String
  studentId       String
  
  status          ApplicationStatus @default(APPLIED)
  currentRound    Int      @default(0)
  
  offerStatus     OfferStatus?
  offerDetails    Json?
  
  drive           CampusDrive @relation(fields: [driveId], references: [id])
  student         User     @relation(fields: [studentId], references: [id])
  
  schedules       InterviewSchedule[]
  
  appliedAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  
  @@unique([driveId, studentId])
}

enum ApplicationStatus {
  APPLIED
  SHORTLISTED
  IN_PROCESS
  REJECTED
  OFFERED
  ACCEPTED
  DECLINED
}

enum OfferStatus {
  PENDING
  ACCEPTED
  DECLINED
  EXPIRED
}

// Gamification
model Achievement {
  id              String   @id @default(cuid())
  code            String   @unique
  name            String
  description     String
  icon            String
  category        AchievementCategory
  
  criteria        Json     // {type: 'count', target: 5, condition: 'assignments_early'}
  xpReward        Int
  creditReward    Int      @default(0)
  
  userAchievements UserAchievement[]
  
  createdAt       DateTime @default(now())
}

enum AchievementCategory {
  CONSISTENCY
  PERFORMANCE
  SPEED
  COLLABORATION
  MILESTONE
}

model UserAchievement {
  id              String   @id @default(cuid())
  userId          String
  achievementId   String
  
  progress        Int      @default(0)
  completed       Boolean  @default(false)
  completedAt     DateTime?
  
  user            User     @relation(fields: [userId], references: [id])
  achievement     Achievement @relation(fields: [achievementId], references: [id])
  
  @@unique([userId, achievementId])
}

model UserStreak {
  id              String   @id @default(cuid())
  userId          String
  type            StreakType
  
  currentStreak   Int      @default(0)
  longestStreak   Int      @default(0)
  lastActiveDate  DateTime
  freezesUsed     Int      @default(0)
  freezesAvailable Int     @default(2)
  
  user            User     @relation(fields: [userId], references: [id])
  
  @@unique([userId, type])
}

enum StreakType {
  DAILY_LOGIN
  ASSIGNMENT_COMPLETION
  PRACTICE
}
