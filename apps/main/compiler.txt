================================================================================
                            1. OVERVIEW & ARCHITECTURE
================================================================================

The Coder'z Compiler is a standalone Node.js + Express + TypeScript backend 
service that executes user-submitted code in a secure, sandboxed environment.
It supports multiple programming languages and implements safeguards against
malicious code, infinite loops, and resource exhaustion.

KEY FEATURES:
- Multi-language support (C, C++, JavaScript, Python, TypeScript, Java)
- Secure Docker-based sandboxing
- Infinite loop & resource exhaustion protection
- LeetCode-style polling mechanism
- Horizontal scalability for 1000s of concurrent users
- Isolated execution per submission

export type Language = 'c' | 'cpp' | 'javascript' | 'python' | 'typescript' | 'java';

export type SubmissionStatus = 
  | 'pending'      // In queue
  | 'running'      // Currently executing
  | 'completed'    // Successfully completed
  | 'error'        // Compilation/Runtime error
  | 'timeout'      // Exceeded time limit
  | 'memory_limit' // Exceeded memory limit

export interface SubmissionRequest {
  code: string;
  language: Language;
  testCases?: TestCase[];
  problemDescription?: string;
  stdin?: string;
}

export interface TestCase {
  input: string;
  expectedOutput: string;
}

export interface SubmissionResult {
  id: string;
  status: SubmissionStatus;
  output?: string;
  error?: string;
  executionTime?: number;  // in milliseconds
  memoryUsed?: number;     // in KB
  testResults?: TestCaseResult[];
  createdAt: Date;
  completedAt?: Date;
}

export interface TestCaseResult {
  passed: boolean;
  input: string;
  expectedOutput: string;
  actualOutput: string;
  executionTime: number;
}

export interface LanguageConfig {
  name: string;
  extension: string;
  dockerImage: string;
  compileCommand?: string;
  runCommand: string;
  timeout: number;      // in seconds
  memoryLimit: number;  // in MB
}

// ============= src/controllers/submission.controller.ts =============

import { Router, Request, Response } from 'express';
import { v4 as uuidv4 } from 'uuid';
import { submissionQueue } from '../services/queue.service';
import { getSubmissionResult } from '../services/storage.service';
import { validateSubmission } from '../middlewares/validation.middleware';
import { SubmissionRequest } from '../types';

export const submissionRouter = Router();

// POST /api/submissions - Submit code for execution
submissionRouter.post('/', validateSubmission, async (req: Request, res: Response) => {
  try {
    const { code, language, testCases, stdin } = req.body as SubmissionRequest;
    
    // Generate unique submission ID
    const submissionId = uuidv4();
    
    // Add to queue
    await submissionQueue.add('execute', {
      id: submissionId,
      code,
      language,
      testCases,
      stdin,
      createdAt: new Date(),
    }, {
      jobId: submissionId,
      removeOnComplete: 100,  // Keep last 100 completed jobs
      removeOnFail: 100,
    });
    
    // Return immediately with submission ID (polling pattern)
    res.status(202).json({
      success: true,
      submissionId,
      message: 'Code submitted successfully. Poll /status/:id for results.',
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: 'Failed to submit code',
    });
  }
});

// GET /api/submissions/status/:id - Poll for submission status
submissionRouter.get('/status/:id', async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    
    // Check job status in queue
    const job = await submissionQueue.getJob(id);
    
    if (!job) {
      // Check storage for completed job
      const result = await getSubmissionResult(id);
      if (result) {
        return res.json({ success: true, ...result });
      }
      return res.status(404).json({
        success: false,
        error: 'Submission not found',
      });
    }
    
    const state = await job.getState();
    
    if (state === 'completed') {
      const result = job.returnvalue;
      return res.json({ success: true, ...result });
    }
    
    if (state === 'failed') {
      return res.json({
        success: true,
        status: 'error',
        error: job.failedReason || 'Execution failed',
      });
    }
    
    // Still processing
    return res.json({
      success: true,
      status: state === 'active' ? 'running' : 'pending',
      message: `Submission is ${state}`,
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: 'Failed to get submission status',
    });
  }
});

// ============= src/workers/executor.worker.ts =============

import { Worker, Job } from 'bullmq';
import Redis from 'ioredis';
import { executeCode } from '../services/executor.service';
import { saveSubmissionResult } from '../services/storage.service';
import { logger } from '../utils/logger';

const connection = new Redis(process.env.REDIS_URL || 'redis://localhost:6379', {
  maxRetriesPerRequest: null,
});

export function startWorker() {
  const worker = new Worker(
    'submissions',
    async (job: Job) => {
      const { id, code, language, testCases, stdin } = job.data;
      
      logger.info(`Processing submission ${id} for language ${language}`);
      
      try {
        const result = await executeCode({
          code,
          language,
          testCases,
          stdin,
        });
        
        const submissionResult = {
          id,
          ...result,
          completedAt: new Date(),
        };
        
        // Save to storage
        await saveSubmissionResult(id, submissionResult);
        
        return submissionResult;
      } catch (error) {
        logger.error(`Execution failed for ${id}:`, error);
        throw error;
      }
    },
    {
      connection,
      concurrency: parseInt(process.env.WORKER_CONCURRENCY || '5'),
    }
  );
  
  worker.on('completed', (job) => {
    logger.info(`Job ${job.id} completed`);
  });
  
  worker.on('failed', (job, err) => {
    logger.error(`Job ${job?.id} failed:`, err);
  });
  
  logger.info('Executor worker started');
}


// ============= src/services/executor.service.ts =============

import Docker from 'dockerode';
import { v4 as uuidv4 } from 'uuid';
import * as fs from 'fs/promises';
import * as path from 'path';
import { LANGUAGE_CONFIGS } from '../config/languages';
import { Language, SubmissionResult, TestCase, TestCaseResult } from '../types';
import { logger } from '../utils/logger';

const docker = new Docker();

interface ExecuteOptions {
  code: string;
  language: Language;
  testCases?: TestCase[];
  stdin?: string;
}

export async function executeCode(options: ExecuteOptions): Promise<Partial<SubmissionResult>> {
  const { code, language, testCases, stdin } = options;
  const config = LANGUAGE_CONFIGS[language];
  const executionId = uuidv4();
  const workDir = `/tmp/coderz/${executionId}`;
  
  try {
    // Create temp directory
    await fs.mkdir(workDir, { recursive: true });
    
    // Write code to file
    const fileName = language === 'java' ? 'Main.java' : `code${config.extension}`;
    await fs.writeFile(path.join(workDir, fileName), code);
    
    // Write stdin if provided
    if (stdin) {
      await fs.writeFile(path.join(workDir, 'input.txt'), stdin);
    }
    
    const startTime = Date.now();
    
    // Create container with strict limits
    const container = await docker.createContainer({
      Image: config.dockerImage,
      Cmd: ['/bin/sh', '-c', buildExecutionCommand(config, language, stdin)],
      HostConfig: {
        Memory: config.memoryLimit * 1024 * 1024,  // Convert MB to bytes
        MemorySwap: config.memoryLimit * 1024 * 1024,
        CpuPeriod: 100000,
        CpuQuota: 100000,  // 1 CPU
        NetworkMode: 'none',  // No network access
        ReadonlyRootfs: false,
        Binds: [`${workDir}:/tmp:rw`],
        AutoRemove: true,
      },
      WorkingDir: '/tmp',
      AttachStdout: true,
      AttachStderr: true,
      Tty: false,
    });
    
    await container.start();
    
    // Wait for completion with timeout
    const result = await waitForContainer(container, config.timeout * 1000);
    
    const executionTime = Date.now() - startTime;
    
    // Process test cases if provided
    if (testCases && testCases.length > 0) {
      const testResults = await runTestCases(options, testCases);
      return {
        status: testResults.every(t => t.passed) ? 'completed' : 'error',
        output: result.stdout,
        error: result.stderr || undefined,
        executionTime,
        testResults,
        createdAt: new Date(),
      };
    }
    
    return {
      status: result.timedOut ? 'timeout' : (result.exitCode === 0 ? 'completed' : 'error'),
      output: result.stdout,
      error: result.stderr || undefined,
      executionTime,
      createdAt: new Date(),
    };
    
  } catch (error) {
    logger.error('Execution error:', error);
    return {
      status: 'error',
      error: error instanceof Error ? error.message : 'Unknown error',
      createdAt: new Date(),
    };
  } finally {
    // Cleanup
    await fs.rm(workDir, { recursive: true, force: true }).catch(() => {});
  }
}

function buildExecutionCommand(config: typeof LANGUAGE_CONFIGS[Language], language: Language, stdin?: string): string {
  let command = '';
  
  // Compile if needed
  if (config.compileCommand) {
    command += `${config.compileCommand} && `;
  }
  
  // Run
  if (stdin) {
    command += `${config.runCommand} < /tmp/input.txt`;
  } else {
    command += config.runCommand;
  }
  
  return command;
}

async function waitForContainer(
  container: Docker.Container, 
  timeout: number
): Promise<{ stdout: string; stderr: string; exitCode: number; timedOut: boolean }> {
  return new Promise(async (resolve) => {
    const timeoutId = setTimeout(async () => {
      try {
        await container.kill();
      } catch {}
      resolve({ stdout: '', stderr: 'Execution timed out', exitCode: -1, timedOut: true });
    }, timeout);
    
    try {
      const stream = await container.attach({
        stream: true,
        stdout: true,
        stderr: true,
      });
      
      let stdout = '';
      let stderr = '';
      
      container.modem.demuxStream(stream, {
        write: (chunk: Buffer) => { stdout += chunk.toString(); }
      }, {
        write: (chunk: Buffer) => { stderr += chunk.toString(); }
      });
      
      const { StatusCode } = await container.wait();
      clearTimeout(timeoutId);
      
      resolve({ stdout: stdout.trim(), stderr: stderr.trim(), exitCode: StatusCode, timedOut: false });
    } catch (error) {
      clearTimeout(timeoutId);
      resolve({ stdout: '', stderr: 'Container execution failed', exitCode: -1, timedOut: false });
    }
  });
}

async function runTestCases(options: ExecuteOptions, testCases: TestCase[]): Promise<TestCaseResult[]> {
  const results: TestCaseResult[] = [];
  
  for (const testCase of testCases) {
    const startTime = Date.now();
    const result = await executeCode({
      ...options,
      stdin: testCase.input,
      testCases: undefined,
    });
    
    const actualOutput = result.output?.trim() || '';
    const expectedOutput = testCase.expectedOutput.trim();
    
    results.push({
      passed: actualOutput === expectedOutput,
      input: testCase.input,
      expectedOutput,
      actualOutput,
      executionTime: Date.now() - startTime,
    });
  }
  
  return results;
}


// ============= src/services/storage.service.ts =============

import Redis from 'ioredis';
import { SubmissionResult } from '../types';

const redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379');

const RESULT_TTL = 3600; // 1 hour

export async function saveSubmissionResult(id: string, result: SubmissionResult): Promise<void> {
  await redis.setex(`submission:${id}`, RESULT_TTL, JSON.stringify(result));
}

export async function getSubmissionResult(id: string): Promise<SubmissionResult | null> {
  const data = await redis.get(`submission:${id}`);
  return data ? JSON.parse(data) : null;
}

================================================================================
                            7. SECURITY & SANDBOXING
================================================================================

SECURITY MEASURES:
------------------

1. DOCKER ISOLATION:
   - Each submission runs in its own container
   - Containers are destroyed after execution
   - No persistent state between runs

2. NETWORK ISOLATION:
   - NetworkMode: 'none' - No internet access
   - Prevents data exfiltration
   - Prevents external attacks

3. RESOURCE LIMITS:

Input: while(true) printf("%d", 1);

================================================================================
                            9. POLLING MECHANISM (LEETCODE-STYLE)
================================================================================

WHY POLLING?
------------
- Prevents long-running HTTP connections
- Better scalability with 1000s of users
- Simpler infrastructure (no WebSockets initially)
- Client can retry if connection drops

FLOW:
-----

1. Client submits code → Receives submission_id immediately
2. Client polls /status/:id every 1-2 seconds
3. Server returns current status (pending/running/completed/error)
4. Client stops polling when status is final


// Frontend Polling Example:

async function submitAndPoll(code: string, language: string) {
  // 1. Submit code
  const submitRes = await fetch('/api/compiler/submit', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ code, language }),
  });
  const { submissionId } = await submitRes.json();
  
  // 2. Poll for results
  let result = null;
  while (true) {
    const statusRes = await fetch(`/api/compiler/status/${submissionId}`);
    result = await statusRes.json();
    
    if (['completed', 'error', 'timeout'].includes(result.status)) {
      break;
    }
    
    // Wait 1 second before next poll
    await new Promise(r => setTimeout(r, 1000));
  }
  
  return result;
}
